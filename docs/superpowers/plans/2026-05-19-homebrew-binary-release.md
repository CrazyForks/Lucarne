# Homebrew Binary Release Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make `brew install lucarned` download macOS Apple Silicon and Intel release binaries instead of rebuilding Rust locally.

**Architecture:** GitHub Actions builds architecture-specific `lucarned` tarballs on macOS runners, publishes them to GitHub Releases, tests the generated Homebrew formula on both macOS architectures, then commits the binary formula update back to `main`. A small Python updater generates `Formula/lucarned.rb` deterministically from release version and SHA-256 values.

**Tech Stack:** Rust nightly Cargo, GitHub Actions, Homebrew Formula Ruby DSL, Python standard library `unittest`, macOS `otool`.

---

## File Structure

- Create: `scripts/release/update-homebrew-formula.py`
  - Deterministically writes `Formula/lucarned.rb` for binary releases.
  - Validates semantic-ish version strings and 64-character lowercase SHA-256 values.
  - Keeps source build only under `head do` and `if build.head?`.
- Create: `scripts/release/test_update_homebrew_formula.py`
  - Unit-tests formula generation and input validation.
- Create: `.github/workflows/release.yml`
  - Builds `aarch64-apple-darwin` on `macos-14-arm64`.
  - Builds `x86_64-apple-darwin` on `macos-13`.
  - Publishes GitHub Release tarballs and `checksums.txt`.
  - Tests generated formula on both macOS architectures.
  - Updates `Formula/lucarned.rb` on `main` after formula tests pass.
- Modify after first release job succeeds: `Formula/lucarned.rb`
  - Generated by `scripts/release/update-homebrew-formula.py` with real release URLs and SHA-256 values.

#!/usr/bin/env python3
from __future__ import annotations

import subprocess
import sys
import tempfile
import textwrap
import unittest
from pathlib import Path

SCRIPT = Path(__file__).with_name("update-homebrew-formula.py")


class UpdateHomebrewFormulaTest(unittest.TestCase):
    def run_script(
        self,
        source: str,
        version: str = "0.2.3",
        arm64_sha: str = "a" * 64,
        x86_64_sha: str = "b" * 64,
    ) -> str:
        with tempfile.TemporaryDirectory() as tmp:
            formula = Path(tmp) / "lucarned.rb"
            formula.write_text(source, encoding="utf-8")
            subprocess.run(
                [
                    sys.executable,
                    str(SCRIPT),
                    str(formula),
                    "--version",
                    version,
                    "--arm64-sha",
                    arm64_sha,
                    "--x86-64-sha",
                    x86_64_sha,
                ],
                check=True,
                text=True,
            )
            return formula.read_text(encoding="utf-8")

    def test_rewrites_source_formula_to_binary_formula(self) -> None:
        source = textwrap.dedent(
            """
            class Lucarned < Formula
              desc "old"
              homepage "https://example.invalid"
              url "https://github.com/tuchg/Lucarne.git", branch: "main"

              depends_on "rust" => :build

              def install
                system "cargo", "install"
              end
            end
            """
        ).lstrip()

        output = self.run_script(source)

        self.assertIn('class Lucarned < Formula', output)
        self.assertIn('desc "Local lucarne daemon for channel adapters and agent sessions"', output)
        self.assertIn('homepage "https://github.com/tuchg/Lucarne"', output)
        self.assertIn('version "0.2.3"', output)
        self.assertIn('license "MIT"', output)
        self.assertIn('depends_on :macos', output)
        self.assertIn('on_arm do', output)
        self.assertIn(
            'url "https://github.com/tuchg/Lucarne/releases/download/v0.2.3/lucarned-v0.2.3-aarch64-apple-darwin.tar.gz"',
            output,
        )
        self.assertIn('sha256 "' + "a" * 64 + '"', output)
        self.assertIn('on_intel do', output)
        self.assertIn(
            'url "https://github.com/tuchg/Lucarne/releases/download/v0.2.3/lucarned-v0.2.3-x86_64-apple-darwin.tar.gz"',
            output,
        )
        self.assertIn('sha256 "' + "b" * 64 + '"', output)
        self.assertIn('head do', output)
        self.assertIn('depends_on "pkg-config" => :build', output)
        self.assertIn('depends_on "rust" => :build', output)
        self.assertIn('depends_on "openssl@3"', output)
        self.assertIn('if build.head?', output)
        self.assertIn('bin.install "bin/lucarned"', output)
        self.assertIn('brew services start lucarned', output)
        self.assertIn('assert_match "enabled: false", config.read', output)
        self.assertNotIn('url "https://github.com/tuchg/Lucarne.git", branch: "main"\n  version', output)

    def test_rejects_bad_arm64_sha(self) -> None:
        source = "class Lucarned < Formula\nend\n"
        with tempfile.TemporaryDirectory() as tmp:
            formula = Path(tmp) / "lucarned.rb"
            formula.write_text(source, encoding="utf-8")
            result = subprocess.run(
                [
                    sys.executable,
                    str(SCRIPT),
                    str(formula),
                    "--version",
                    "0.2.3",
                    "--arm64-sha",
                    "not-a-sha",
                    "--x86-64-sha",
                    "b" * 64,
                ],
                text=True,
                capture_output=True,
            )

        self.assertNotEqual(result.returncode, 0)
        self.assertIn("--arm64-sha must be 64 lowercase hex characters", result.stderr)

    def test_rejects_bad_version(self) -> None:
        source = "class Lucarned < Formula\nend\n"
        with tempfile.TemporaryDirectory() as tmp:
            formula = Path(tmp) / "lucarned.rb"
            formula.write_text(source, encoding="utf-8")
            result = subprocess.run(
                [
                    sys.executable,
                    str(SCRIPT),
                    str(formula),
                    "--version",
                    "release-0.2.3",
                    "--arm64-sha",
                    "a" * 64,
                    "--x86-64-sha",
                    "b" * 64,
                ],
                text=True,
                capture_output=True,
            )

        self.assertNotEqual(result.returncode, 0)
        self.assertIn("--version must look like 0.1.0", result.stderr)


if __name__ == "__main__":
    unittest.main()

Create `scripts/release/update-homebrew-formula.py` with this content:

```python
#!/usr/bin/env python3
from __future__ import annotations

import argparse
import re
import sys
from pathlib import Path

SHA256_RE = re.compile(r"^[0-9a-f]{64}$")
VERSION_RE = re.compile(r"^\d+\.\d+\.\d+(?:[-+][0-9A-Za-z.-]+)?$")


def die(message: str) -> None:
    print(message, file=sys.stderr)
    raise SystemExit(2)


def validate_sha(flag: str, value: str) -> str:
    if not SHA256_RE.fullmatch(value):
        die(f"{flag} must be 64 lowercase hex characters")
    return value


def validate_version(value: str) -> str:
    if not VERSION_RE.fullmatch(value):
        die("--version must look like 0.1.0")
    return value


def formula_text(version: str, arm64_sha: str, x86_64_sha: str) -> str:
    arm64_url = (
        "https://github.com/tuchg/Lucarne/releases/download/"
        f"v{version}/lucarned-v{version}-aarch64-apple-darwin.tar.gz"
    )
    x86_64_url = (
        "https://github.com/tuchg/Lucarne/releases/download/"
        f"v{version}/lucarned-v{version}-x86_64-apple-darwin.tar.gz"
    )
    return f'''class Lucarned < Formula
  desc "Local lucarne daemon for channel adapters and agent sessions"
  homepage "https://github.com/tuchg/Lucarne"
  version "{version}"
  license "MIT"

  depends_on :macos

  stable do
    on_macos do
      on_arm do
        url "{arm64_url}"
        sha256 "{arm64_sha}"
      end

      on_intel do
        url "{x86_64_url}"
        sha256 "{x86_64_sha}"
      end
    end
  end

  head do
    url "https://github.com/tuchg/Lucarne.git", branch: "main"

    depends_on "pkg-config" => :build
    depends_on "rust" => :build
    depends_on "openssl@3"
  end

  def install
    if build.head?
      ENV["OPENSSL_DIR"] = Formula["openssl@3"].opt_prefix

      # Release builds should consume the published wechat-ilink crate instead of
      # the repository-local development patch path.
      inreplace "Cargo.toml", %r{{^\[patch\.crates-io\]\nwechat-ilink = \{{ path = "\.\./wechat-ilink" \}}\n}}m, ""

      system "cargo", "install", "--path", "crates/lucarned", "--root", prefix, "--no-track"
    else
      bin.install "bin/lucarned"
    end
  end

  service do
    run [opt_bin/"lucarned"]
    keep_alive false
    log_path var/"log/lucarned/brew.out.log"
    error_log_path var/"log/lucarned/brew.err.log"
    working_dir HOMEBREW_PREFIX
  end

  def caveats
    <<~EOS
      lucarned creates ~/.lucarned/lucarned.yaml during setup.

      Basic setup:
        lucarned init
        brew services start lucarned

      lucarned init is interactive; run it in a terminal. It can validate
      Telegram settings and show a WeChat QR code login.

      Config can enable selected agents (omit agents to enable all compiled agents):
        agents:
          - codex
          - pi

      Config can enable channels before starting service, for example Telegram:
        channels:
          telegram:
            enabled: true
            token: "123456:REDACTED"
            entry_chat_id: 123456789

      Common commands:
        brew services start lucarned
        brew services stop lucarned
        brew services restart lucarned

      Logs:   ~/.lucarned/logs/lucarned.YYYY-MM-DD.log
      Config: ~/.lucarned/lucarned.yaml
      State:  ~/.lucarned/state.sqlite3
      Brew service logs:
        #{{var}}/log/lucarned/brew.out.log
        #{{var}}/log/lucarned/brew.err.log
    EOS
  end

  test do
    ENV["HOME"] = testpath
    ENV.delete "LUCARNE_CONFIG"
    ENV.delete "LUCARNED_CONFIG"
    ENV.delete "LUCARNE_STATE_DB"
    ENV.delete "LUCARNE_LOG_FILE"

    system bin/"lucarned"
    config = testpath/".lucarned/lucarned.yaml"
    assert_path_exists config
    assert_match "agents:", config.read
    assert_match "telegram:", config.read
    assert_match "wechat:", config.read
    assert_match "enabled: false", config.read
  end
end
'''


def main() -> int:
    parser = argparse.ArgumentParser(description="Update Formula/lucarned.rb for a binary release")
    parser.add_argument("formula", type=Path)
    parser.add_argument("--version", required=True)
    parser.add_argument("--arm64-sha", required=True)
    parser.add_argument("--x86-64-sha", required=True)
    args = parser.parse_args()

    version = validate_version(args.version)
    arm64_sha = validate_sha("--arm64-sha", args.arm64_sha)
    x86_64_sha = validate_sha("--x86-64-sha", args.x86_64_sha)

    if not args.formula.exists():
        die(f"formula file does not exist: {args.formula}")

    args.formula.write_text(formula_text(version, arm64_sha, x86_64_sha), encoding="utf-8")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

- [ ] **Step 1: Create release workflow**

Create `.github/workflows/release.yml` with this content:

```yaml
name: Release lucarned

on:
  push:
    tags:
      - "v*"
  workflow_dispatch:
    inputs:
      tag:
        description: "Release tag such as v0.1.0"
        required: true
        type: string

permissions:
  contents: write

env:
  CARGO_TERM_COLOR: always

jobs:
  build:
    name: Build ${{ matrix.target }}
    runs-on: ${{ matrix.runner }}
    strategy:
      fail-fast: false
      matrix:
        include:
          - target: aarch64-apple-darwin
            runner: macos-14-arm64
          - target: x86_64-apple-darwin
            runner: macos-13
    steps:
      - name: Resolve release tag
        id: release
        shell: bash
        run: |
          set -euo pipefail
          TAG="${{ github.event.inputs.tag || github.ref_name }}"
          if [[ ! "$TAG" =~ ^v[0-9]+\.[0-9]+\.[0-9]+([-+][0-9A-Za-z.-]+)?$ ]]; then
            echo "Invalid release tag: $TAG" >&2
            exit 1
          fi
          echo "tag=$TAG" >> "$GITHUB_OUTPUT"
          echo "version=${TAG#v}" >> "$GITHUB_OUTPUT"

      - name: Checkout release ref
        uses: actions/checkout@v4
        with:
          ref: ${{ steps.release.outputs.tag }}

      - name: Validate Cargo version matches tag
        shell: bash
        run: |
          set -euo pipefail
          CARGO_VERSION="$(python3 - <<'PY'
import tomllib
from pathlib import Path
with Path('Cargo.toml').open('rb') as f:
    cargo = tomllib.load(f)
print(cargo['workspace']['package']['version'])
PY
)"
          if [ "$CARGO_VERSION" != "${{ steps.release.outputs.version }}" ]; then
            echo "Cargo version $CARGO_VERSION does not match tag ${{ steps.release.outputs.tag }}" >&2
            exit 1
          fi

      - name: Install Rust nightly
        shell: bash
        run: |
          set -euo pipefail
          rustup toolchain install nightly --profile minimal

      - name: Prepare release workspace
        shell: bash
        run: |
          set -euo pipefail
          python3 - <<'PY'
from pathlib import Path
path = Path('Cargo.toml')
text = path.read_text()
patch = '[patch.crates-io]\nwechat-ilink = { path = "../wechat-ilink" }\n'
if patch not in text:
    raise SystemExit('expected wechat-ilink patch block not found')
path.write_text(text.replace(patch, ''))
PY
          cargo +nightly update -Zbuild-dir-new-layout -p wechat-ilink --precise 0.5.0

      - name: Build lucarned
        shell: bash
        run: |
          set -euo pipefail
          cargo +nightly build --release -Zbuild-dir-new-layout --locked -p lucarned

      - name: Locate built binary
        id: binary
        shell: bash
        run: |
          set -euo pipefail
          BIN="$(find target -type f -path '*/release/lucarned' -perm -111 | head -n 1)"
          if [ -z "$BIN" ]; then
            echo "Could not find release lucarned binary" >&2
            find target -type f -name lucarned -print >&2
            exit 1
          fi
          echo "path=$BIN" >> "$GITHUB_OUTPUT"
          "$BIN" --help >/dev/null

      - name: Smoke test binary
        shell: bash
        run: |
          set -euo pipefail
          TMP_HOME="$(mktemp -d)"
          HOME="$TMP_HOME" "${{ steps.binary.outputs.path }}"
          test -f "$TMP_HOME/.lucarned/lucarned.yaml"
          grep -q '^agents:' "$TMP_HOME/.lucarned/lucarned.yaml"
          grep -q '^channels:' "$TMP_HOME/.lucarned/lucarned.yaml"

      - name: Check dynamic libraries
        shell: bash
        run: |
          set -euo pipefail
          otool -L "${{ steps.binary.outputs.path }}" | tee otool.txt
          if grep -E '(/opt/homebrew|/usr/local/(Cellar|opt))' otool.txt; then
            echo "Unexpected Homebrew dynamic library dependency" >&2
            exit 1
          fi

      - name: Package tarball
        shell: bash
        run: |
          set -euo pipefail
          TAG="${{ steps.release.outputs.tag }}"
          TARGET="${{ matrix.target }}"
          DIST="$PWD/dist"
          STAGE="$RUNNER_TEMP/lucarned-stage"
          rm -rf "$DIST" "$STAGE"
          mkdir -p "$DIST" "$STAGE/bin"
          cp "${{ steps.binary.outputs.path }}" "$STAGE/bin/lucarned"
          cp LICENSE README.md "$STAGE/"
          chmod 755 "$STAGE/bin/lucarned"
          TAR="$DIST/lucarned-$TAG-$TARGET.tar.gz"
          tar -czf "$TAR" -C "$STAGE" .
          tar -tzf "$TAR" | sort | tee tarball-files.txt
          grep -qx './bin/lucarned' tarball-files.txt
          grep -qx './LICENSE' tarball-files.txt
          grep -qx './README.md' tarball-files.txt

      - name: Upload tarball artifact
        uses: actions/upload-artifact@v4
        with:
          name: lucarned-${{ matrix.target }}
          path: dist/*.tar.gz
          if-no-files-found: error

  publish-assets:
    name: Publish release assets
    runs-on: ubuntu-latest
    needs: build
    outputs:
      version: ${{ steps.metadata.outputs.version }}
      tag: ${{ steps.metadata.outputs.tag }}
      arm64_sha: ${{ steps.metadata.outputs.arm64_sha }}
      x86_64_sha: ${{ steps.metadata.outputs.x86_64_sha }}
    steps:
      - name: Resolve release tag
        id: release
        shell: bash
        run: |
          set -euo pipefail
          TAG="${{ github.event.inputs.tag || github.ref_name }}"
          if [[ ! "$TAG" =~ ^v[0-9]+\.[0-9]+\.[0-9]+([-+][0-9A-Za-z.-]+)?$ ]]; then
            echo "Invalid release tag: $TAG" >&2
            exit 1
          fi
          echo "tag=$TAG" >> "$GITHUB_OUTPUT"
          echo "version=${TAG#v}" >> "$GITHUB_OUTPUT"

      - name: Checkout release ref
        uses: actions/checkout@v4
        with:
          ref: ${{ steps.release.outputs.tag }}

      - name: Download tarball artifacts
        uses: actions/download-artifact@v4
        with:
          path: dist
          pattern: lucarned-*
          merge-multiple: true

      - name: Build checksum metadata
        id: metadata
        shell: bash
        run: |
          set -euo pipefail
          cd dist
          sha256sum *.tar.gz | sort -k2 > checksums.txt
          cat checksums.txt
          ARM64_SHA="$(awk '/aarch64-apple-darwin/ {print $1}' checksums.txt)"
          X86_64_SHA="$(awk '/x86_64-apple-darwin/ {print $1}' checksums.txt)"
          if [ -z "$ARM64_SHA" ] || [ -z "$X86_64_SHA" ]; then
            echo "Missing architecture checksum" >&2
            exit 1
          fi
          cd ..
          cat > release-metadata.env <<EOF
VERSION=${{ steps.release.outputs.version }}
TAG=${{ steps.release.outputs.tag }}
ARM64_SHA=$ARM64_SHA
X86_64_SHA=$X86_64_SHA
EOF
          echo "version=${{ steps.release.outputs.version }}" >> "$GITHUB_OUTPUT"
          echo "tag=${{ steps.release.outputs.tag }}" >> "$GITHUB_OUTPUT"
          echo "arm64_sha=$ARM64_SHA" >> "$GITHUB_OUTPUT"
          echo "x86_64_sha=$X86_64_SHA" >> "$GITHUB_OUTPUT"

      - name: Upload release metadata artifact
        uses: actions/upload-artifact@v4
        with:
          name: release-metadata
          path: release-metadata.env
          if-no-files-found: error

      - name: Create or update GitHub Release
        env:
          GH_TOKEN: ${{ github.token }}
        shell: bash
        run: |
          set -euo pipefail
          TAG="${{ steps.release.outputs.tag }}"
          if gh release view "$TAG" >/dev/null 2>&1; then
            gh release upload "$TAG" dist/*.tar.gz dist/checksums.txt --clobber
          else
            gh release create "$TAG" dist/*.tar.gz dist/checksums.txt --title "$TAG" --notes "lucarned $TAG" --verify-tag
          fi

  formula-test:
    name: Test formula ${{ matrix.name }}
    runs-on: ${{ matrix.runner }}
    needs: publish-assets
    strategy:
      fail-fast: false
      matrix:
        include:
          - name: macOS Apple Silicon
            runner: macos-14-arm64
          - name: macOS Intel
            runner: macos-13
    steps:
      - name: Checkout release ref
        uses: actions/checkout@v4
        with:
          ref: ${{ needs.publish-assets.outputs.tag }}

      - name: Download release metadata
        uses: actions/download-artifact@v4
        with:
          name: release-metadata
          path: .

      - name: Generate binary formula
        shell: bash
        run: |
          set -euo pipefail
          source release-metadata.env
          python3 scripts/release/update-homebrew-formula.py Formula/lucarned.rb \
            --version "$VERSION" \
            --arm64-sha "$ARM64_SHA" \
            --x86-64-sha "$X86_64_SHA"
          ruby -c Formula/lucarned.rb

      - name: Create temporary tap
        shell: bash
        run: |
          set -euo pipefail
          brew tap-new tuchg/lucarne --no-git
          cp Formula/lucarned.rb "$(brew --repo tuchg/lucarne)/Formula/lucarned.rb"

      - name: Verify stable formula does not depend on Rust build stack
        shell: bash
        run: |
          set -euo pipefail
          DEPS="$(brew deps --include-build tuchg/lucarne/lucarned)"
          printf '%s\n' "$DEPS"
          if printf '%s\n' "$DEPS" | grep -E '^(rust|pkg-config|openssl@3)$'; then
            echo "Stable binary formula must not depend on Rust, pkg-config, or openssl@3" >&2
            exit 1
          fi

      - name: Install and test formula
        shell: bash
        run: |
          set -euo pipefail
          brew install tuchg/lucarne/lucarned
          brew test tuchg/lucarne/lucarned

  update-formula:
    name: Update formula on main
    runs-on: ubuntu-latest
    needs:
      - publish-assets
      - formula-test
    steps:
      - name: Checkout main
        uses: actions/checkout@v4
        with:
          ref: main

      - name: Download release metadata
        uses: actions/download-artifact@v4
        with:
          name: release-metadata
          path: .

      - name: Generate binary formula
        shell: bash
        run: |
          set -euo pipefail
          source release-metadata.env
          python3 scripts/release/update-homebrew-formula.py Formula/lucarned.rb \
            --version "$VERSION" \
            --arm64-sha "$ARM64_SHA" \
            --x86-64-sha "$X86_64_SHA"
          ruby -c Formula/lucarned.rb

      - name: Commit formula update
        shell: bash
        run: |
          set -euo pipefail
          if git diff --quiet -- Formula/lucarned.rb; then
            echo "Formula already up to date"
            exit 0
          fi
          git config user.name "github-actions[bot]"
          git config user.email "41898282+github-actions[bot]@users.noreply.github.com"
          VERSION="${{ needs.publish-assets.outputs.version }}"
          git add Formula/lucarned.rb
          git commit -m "brew: update lucarned v$VERSION"
          git push origin HEAD:main
```
